perform select with casts

select cast(schluessel as char(6)) as schluessel ,
cast (straftat as varchar (50)) as straftat,
cast (gemeindeschluessel as int) as gmd_schl,
cast (stadt_landkreis as varchar(100)) as kreis_name, 
cast (kreisart as varchar(3)) as kreisart,
cast (replace(anzahl_erfasste_faelle,',','') as int) as anzahl_erfasste_faelle,
cast (replace(haeufigkeitszahl,',','') as int) as haeufigkeitszahl,
cast (replace(erfasste_faelle_versuche_anzahl,',','') as int) as erfasste_faelle_versuche_anzahl,
cast (replace(mit_schusswaffe_gedroht,',','') as int) as mit_schusswaffe_gedroht,
cast (replace(mit_schusswaffe_geschossen,',','') as int) as mit_schusswaffe_geschossen,
cast (replace(auffklaerung_anzahl_faelle,',','') as int) as aufklaerung_anzahl_faelle,
cast (replace(tatverdaechtige_insgesamt,',','') as int) as tatverdaechtige
from kriminalstatistik_2022;
